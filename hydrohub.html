Google Apps Script Setup

To log your water intake to a Google Sheet, follow these steps:

  1. Go to script.google.com and create a new project.
  2. Copy the code below and paste it into the script editor, replacing any existing code.
  3. Click the "Deploy" button, select "New deployment".
  4. For "Select type", choose "Web app".
  5. In the "New deployment" dialog:
    • Give it a description (e.g., "Water Tracker").
    • Set "Execute as" to "Me".
    • Set "Who has access" to "Anyone". Important: This makes your web app public but only able to add data, not read it.
  6. Click "Deploy". You will need to authorize the script to access your Google Sheets.
  7. Copy the "Web app URL" provided and paste it into the settings of this water tracker app.
function doPost(e) {
  try {
    var sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName("WaterIntake");
    if (!sheet) {
      sheet = SpreadsheetApp.getActiveSpreadsheet().insertSheet("WaterIntake");
      sheet.appendRow(["Timestamp", "Amount (ml)", "Total Today (ml)"]);
    }
    
    var data = JSON.parse(e.postData.contents);
    var timestamp = new Date();
    
    sheet.appendRow([timestamp, data.amount, data.total]);
    
    return ContentService.createTextOutput(JSON.stringify({ "status": "success" }))
      .setMimeType(ContentService.MimeType.JSON);
      
  } catch (error) {
    return ContentService.createTextOutput(JSON.stringify({ "status": "error", "message": error.toString() }))
      .setMimeType(ContentService.MimeType.JSON);
  }
}
