Set Up Your Own Comic Library

Follow these steps to create your own private library powered by Google Sheets. It's free and your data stays in your Google account.

Step 1: Create your Google Sheet

Go to sheets.google.com/create to make a new, blank spreadsheet. You can name it whatever you like, for example, "My Comic Library". The app will create the necessary tabs automatically.

Step 2: Create the Backend Script

  1. In your new spreadsheet, click on Extensions → Apps Script.
  2. A new browser tab will open with the script editor. Delete any code that's already there.
  3. Copy the entire block of code below and paste it into the editor.
const SPREADSHEET_ID = "YOUR_SPREADSHEET_ID"; // You can hardcode this or leave it to be inferred
const BOOK_SHEET_NAME = "Books";
const SETTINGS_SHEET_NAME = "_settings";

// Helper to get the active spreadsheet
function getSpreadsheet() {
  if (SPREADSHEET_ID) {
    return SpreadsheetApp.openById(SPREADSHEET_ID);
  }
  return SpreadsheetApp.getActiveSpreadsheet();
}

// Function to set up necessary sheets if they don't exist
function setupSheets() {
  const ss = getSpreadsheet();
  
  // Setup Books Sheet
  let bookSheet = ss.getSheetByName(BOOK_SHEET_NAME);
  if (!bookSheet) {
    bookSheet = ss.insertSheet(BOOK_SHEET_NAME);
    const headers = ["id", "title", "author", "series", "category", "coverImageUrl", "labels", "notes", "dateAdded", "lastUpdated"];
    bookSheet.appendRow(headers);
    bookSheet.setFrozenRows(1);
  }
  
  // Setup Settings Sheet
  let settingsSheet = ss.getSheetByName(SETTINGS_SHEET_NAME);
  if (!settingsSheet) {
    settingsSheet = ss.insertSheet(SETTINGS_SHEET_NAME);
    settingsSheet.appendRow(["Setting", "Value"]);
    settingsSheet.appendRow(["Library Name", "My Comic Library"]); // Default value
    settingsSheet.setFrozenRows(1);
    settingsSheet.hideSheet(); // Hide from user view for cleanliness
  }
}

// Handle GET requests to fetch data
function doGet(e) {
  setupSheets();
  const ss = getSpreadsheet();
  const bookSheet = ss.getSheetByName(BOOK_SHEET_NAME);
  const settingsSheet = ss.getSheetByName(SETTINGS_SHEET_NAME);

  // Get Library Name
  const libraryName = settingsSheet.getRange("B2").getValue();
  
  // Get Books Data
  const bookDataRange = bookSheet.getDataRange();
  const bookValues = bookDataRange.getValues();
  const bookHeaders = bookValues.shift();
  const books = bookValues.map(row => {
    let book = {};
    bookHeaders.forEach((header, index) => {
      book[header] = row[index];
    });
    return book;
  });
  
  const responseData = {
    libraryName: libraryName,
    books: books
  };
  
  return ContentService
    .createTextOutput(JSON.stringify(responseData))
    .setMimeType(ContentService.MimeType.JSON);
}


// Handle POST requests to create, update, delete data, and update settings
function doPost(e) {
  setupSheets();
  const ss = getSpreadsheet();
  const bookSheet = ss.getSheetByName(BOOK_SHEET_NAME);
  const settingsSheet = ss.getSheetByName(SETTINGS_SHEET_NAME);
  const lock = LockService.getScriptLock();
  lock.waitLock(30000); // Wait 30 seconds for lock

  try {
    const request = JSON.parse(e.postData.contents);
    const action = request.action;
    const data = request.data;
    
    switch (action) {
      case 'create':
        return handleCreate(bookSheet, data);
      case 'update':
        return handleUpdate(bookSheet, data);
      case 'delete':
        return handleDelete(bookSheet, data);
      case 'bulkDelete':
        return handleBulkDelete(bookSheet, data);
      case 'updateSettings':
        return handleUpdateSettings(settingsSheet, data);
      default:
        return createJsonResponse({ success: false, message: 'Invalid action' });
    }
  } catch (error) {
    return createJsonResponse({ success: false, message: error.toString() });
  } finally {
    lock.releaseLock();
  }
}

// --- Action Handlers ---

function handleCreate(sheet, data) {
  const newId = Utilities.getUuid();
  const timestamp = new Date();
  const newRow = [newId, data.title, data.author, data.series, data.category, data.coverImageUrl, data.labels, data.notes, timestamp, timestamp];
  sheet.appendRow(newRow);
  return createJsonResponse({ success: true, id: newId });
}

function handleUpdate(sheet, data) {
  const headers = sheet.getRange(1, 1, 1, sheet.getLastColumn()).getValues()[0];
  const idColIndex = headers.indexOf('id') + 1;
  const lastUpdatedColIndex = headers.indexOf('lastUpdated') + 1;
  
  const idRange = sheet.getRange(2, idColIndex, sheet.getLastRow() - 1, 1).getValues();
  const rowIndex = idRange.findIndex(row => row[0] === data.id);
  
  if (rowIndex !== -1) {
    const rowToUpdate = rowIndex + 2; // +1 for 0-index, +1 for header row
    const rowData = headers.map(header => data[header] || '');
    rowData[lastUpdatedColIndex - 1] = new Date(); // Update timestamp
    sheet.getRange(rowToUpdate, 1, 1, headers.length).setValues([rowData]);
    return createJsonResponse({ success: true, id: data.id });
  }
  return createJsonResponse({ success: false, message: 'Book not found' });
}

function handleDelete(sheet, data) {
  const headers = sheet.getRange(1, 1, 1, sheet.getLastColumn()).getValues()[0];
  const idColIndex = headers.indexOf('id') + 1;
  
  const idRange = sheet.getRange(2, idColIndex, sheet.getLastRow() - 1, 1).getValues();
  const rowIndex = idRange.findIndex(row => row[0] === data.id);
  
  if (rowIndex !== -1) {
    const rowToDelete = rowIndex + 2;
    sheet.deleteRow(rowToDelete);
    return createJsonResponse({ success: true, id: data.id });
  }
  return createJsonResponse({ success: false, message: 'Book not found' });
}

function handleBulkDelete(sheet, data) {
  const idsToDelete = data.ids || [];
  if (idsToDelete.length === 0) {
    return createJsonResponse({ success: true, message: 'No IDs provided' });
  }

  const headers = sheet.getRange(1, 1, 1, sheet.getLastColumn()).getValues()[0];
  const idColIndex = headers.indexOf('id') + 1;
  
  const idRange = sheet.getRange(2, idColIndex, sheet.getLastRow() - 1, 1).getValues();
  const rowsToDelete = [];

  idRange.forEach((row, index) => {
    if (idsToDelete.includes(row[0])) {
      rowsToDelete.push(index + 2); // +1 for 0-index, +1 for header
    }
  });

  // Delete rows from the bottom up to avoid index shifting issues
  for (let i = rowsToDelete.length - 1; i >= 0; i--) {
    sheet.deleteRow(rowsToDelete[i]);
  }

  return createJsonResponse({ success: true, deletedCount: rowsToDelete.length });
}


function handleUpdateSettings(sheet, data) {
    if (data.libraryName !== undefined) {
        sheet.getRange("B2").setValue(data.libraryName);
    }
    return createJsonResponse({ success: true, message: 'Settings updated' });
}


// --- Utility ---

function createJsonResponse(obj) {
  return ContentService
    .createTextOutput(JSON.stringify(obj))
    .setMimeType(ContentService.MimeType.JSON);
}

Step 3: Deploy the Script

  1. In the Apps Script editor, click the blue Deploy button, then select New deployment.
  2. Click the gear icon (⚙️) next to "Select type" and choose Web app.
  3. Under "Configuration", set "Who has access" to Anyone. This is important!
  4. Click Deploy. You will need to authorize the script to access your spreadsheet. Follow the prompts.
  5. Once it's deployed, copy the Web app URL. This is the key to your library.

Step 4: Connect it to the App

You're all set! Go back to the main library page and paste your Web app URL into the settings panel.
